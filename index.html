create table if not exists public.jobs ( id text primary key, data jsonb not null default '{}'::jsonb, updated_at timestamptz not null default now() ); alter table public.jobs add column if not exists data jsonb not null default '{}'::jsonb; alter table public.jobs add column if not exists updated_at timestamptz not null default now(); create table if not exists public.admin_tasks ( id text primary key, data jsonb not null default '{}'::jsonb, updated_at timestamptz not null default now() ); alter table public.admin_tasks add column if not exists data jsonb not null default '{}'::jsonb; alter table public.admin_tasks add column if not exists updated_at timestamptz not null default now(); create table if not exists public.system_data ( id text primary key, data jsonb not null default '{}'::jsonb, updated_at timestamptz not null default now() ); alter table public.system_data add column if not exists data jsonb not null default '{}'::jsonb; alter table public.system_data add column if not exists updated_at timestamptz not null default now(); insert into public.jobs (id, data, updated_at) select job->>'id', job, now() from public.app_state s, jsonb_array_elements(coalesce(s.data->'jobs', '[]'::jsonb)) job where s.key = 'service_management_tracker' and coalesce(job->>'id', '') <> '' on conflict (id) do update set data = public.jobs.data || excluded.data, updated_at = now(); insert into public.admin_tasks (id, data, updated_at) select task->>'id', task, now() from public.app_state s, jsonb_array_elements(coalesce(s.data->'adminTasks', '[]'::jsonb)) task where s.key = 'service_management_tracker' and coalesce(task->>'id', '') <> '' on conflict (id) do update set data = public.admin_tasks.data || excluded.data, updated_at = now(); insert into public.system_data (id, data, updated_at) select 'main', data - 'jobs' - 'adminTasks' - 'parts' - 'docs', now() from public.app_state where key = 'service_management_tracker' on conflict (id) do update set data = public.system_data.data || excluded.data, updated_at = now(); select (select count(*) from public.jobs) as jobs_count, (select count(*) from public.admin_tasks) as admin_tasks_count, (select count(*) from public.system_data where id = 'main') as system_data_count;
